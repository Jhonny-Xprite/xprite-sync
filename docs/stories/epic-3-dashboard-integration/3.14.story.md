### Implementation Notes

1. **Backend Service: Hand-off Reader**
   ```typescript
   // packages/api/src/services/handoffs.ts
   import * as fs from 'fs';
   import * as path from 'path';
   import yaml from 'js-yaml';

   class HandoffService {
     async getHandoffHistory(limit = 20): Promise<Handoff[]> {
       const handoffsDir = '.aiox/handoffs';
       const files = fs.readdirSync(handoffsDir)
         .filter(f => f.startsWith('handoff-'))
         .sort()
         .reverse()
         .slice(0, limit);

       return files.map(file => {
         const content = fs.readFileSync(
           path.join(handoffsDir, file),
           'utf-8'
         );
         const data = yaml.load(content);
         return {
           id: file.replace('.yaml', ''),
           ...data.handoff,
         };
       });
     }

     async getHandoff(id: string): Promise<Handoff> {
       const content = fs.readFileSync(
         path.join('.aiox/handoffs', `${id}.yaml`),
         'utf-8'
       );
       return yaml.load(content);
     }
   }
   ```

2. **Backend Endpoints**
   ```typescript
   // packages/api/src/index.ts
   app.get('/api/handoffs/history', async (req, res) => {
     const { limit = 20, from_agent, to_agent } = req.query;
     let handoffs = await handoffService.getHandoffHistory(limit);

     // Filter if specified
     if (from_agent) handoffs = handoffs.filter(h => h.from_agent === from_agent);
     if (to_agent) handoffs = handoffs.filter(h => h.to_agent === to_agent);

     res.json({
       data: handoffs,
       stats: {
         total: handoffs.length,
         success_rate: handoffs.filter(h => h.status === 'completed').length / handoffs.length,
       }
     });
   });

   app.get('/api/handoffs/:id', async (req, res) => {
     const handoff = await handoffService.getHandoff(req.params.id);
     res.json(handoff);
   });
   ```

3. **Frontend Hook**
   ```typescript
   // packages/dashboard/src/hooks/useHandoffs.ts
   export function useHandoffs(limit = 20) {
     return useQuery({
       queryKey: ['handoffs'],
       queryFn: () => handoffsApi.getHistory(limit),
       refetchInterval: 30 * 1000, // 30 seconds
     });
   }

   export function useHandoffDetail(id: string) {
     return useQuery({
       queryKey: ['handoff', id],
       queryFn: () => handoffsApi.getDetail(id),
     });
   }
   ```

4. **Dashboard Component**
   ```typescript
   // packages/dashboard/src/components/handoffs/HandoffMonitor.tsx
   export function HandoffMonitor() {
     const { data: handoffs } = useHandoffs(20);

     if (!handoffs) return <LoadingSkeleton />;

     const lastHandoff = handoffs[0];
     return (
       <div className="space-y-4">
         <div className="text-sm text-gray-500">
           Last hand-off: {lastHandoff.from_agent} → {lastHandoff.to_agent}

         </div>

         <HandoffTimeline handoffs={handoffs} />

         <div className="space-y-2">
           {handoffs.map(h => (
             <HandoffCard key={h.id} handoff={h} />
           ))}
         </div>
       </div>
     );
   }
   ```
