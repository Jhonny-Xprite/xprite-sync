### Implementation Pattern

**For agents service (reference: stories.ts):**
```typescript
export async function getAgents(): Promise<Agent[]> {
  const agentsDir = path.resolve(process.cwd(), '.aiox-core', 'development', 'agents');
  const files = fs.readdirSync(agentsDir).filter(f => f.endsWith('.md'));

  const agents = files.map(file => {
    const content = fs.readFileSync(path.join(agentsDir, file), 'utf-8');
    const yaml = parseYAMLFrontmatter(content);
    return {
      id: yaml.agent.id,
      name: yaml.agent.name,
      title: yaml.agent.title,
      role: yaml.persona.role,
      icon: yaml.agent.icon || '👤',
      status: 'available'
    };
  });

  return agents;
}
```
