```typescript
// Reference: StoriesService.findStoryFiles() pattern
export async function getWorkflows(): Promise<Workflow[]> {
  const workflowsDir = path.resolve(process.cwd(), '.aiox-core', 'development', 'workflows');

  try {
    const files = fs.readdirSync(workflowsDir).filter(f => f.endsWith('.yaml'));
    const workflows = files.map(file => {
      const yaml = parseYAML(fs.readFileSync(path.join(workflowsDir, file), 'utf-8'));
      return {
        id: path.basename(file, '.yaml'),
        name: yaml.name || yaml.title,
        description: yaml.description || '',
        status: 'available',
        phases: yaml.phases?.length || 0,
        agents: yaml.agents || [],
        updated_at: new Date().toISOString()
      };
    });
    return workflows;
  } catch (err) {
    logger.warn('Failed to load workflows', { error: err.message });
    return []; // Graceful fallback
  }
}
```

```typescript
async function checkEngineHealth(): Promise<boolean> {
  try {
    const response = await fetch('http://localhost:4002/health', {
      timeout: 2000 // 2 second timeout
    });
    return response.ok;
  } catch (err) {
    return false; // Timeout or connection error
  }
}

export async function getStatus(): Promise<EngineStatus> {
  const systemMetrics = await systemMetricsCollector.collectCurrentMetrics();
  const engineAvailable = await checkEngineHealth();

  return {
    status: systemMetrics.cpu_percentage < 90 ? 'healthy' : 'degraded',
    pool: engineAvailable
      ? { /* real engine data */ }
      : { active_workers: 0, total_workers: 0, queue_size: 0 },
    health: systemMetrics,
    engine_available: engineAvailable, // HONEST FLAG
    last_check: new Date().toISOString()
  };
}
```

**Phase 3.1: SystemMetricsCollector**
```typescript
export class SystemMetricsCollector {
  constructor(private supabaseService: SupabaseService) {
    this.startPersistence();
  }

  private startPersistence() {
    const teamId = process.env.TEAM_ID;
    if (!teamId) {
      logger.warn('TEAM_ID not set, metrics persistence disabled');
      return;
    }

    setInterval(async () => {
      try {
        const metrics = await this.collectCurrentMetrics();
        await this.supabaseService.insertSystemMetric({
          team_id: teamId,
          ...metrics,
          recorded_at: new Date().toISOString()
        });
      } catch (err) {
        logger.warn('Failed to persist system metrics', { error: err.message });
        // Don't throw — graceful degradation
      }
    }, 60_000); // Every 60 seconds
  }

}
```
